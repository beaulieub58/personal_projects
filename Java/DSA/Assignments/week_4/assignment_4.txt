       tree is correctly maintained and that the balance proporty is in order (pseudo code):

        (Figure 4.37 and 4.38 as guide)
       static AVL class <AnyType> {
        //constructors
        Node <AnyType> {
            this.AnyType, null, null;
        }
        Node <AnyType> (Parameter, LtNode <AnyType>, RtNode <AnyType)  {
            parameter = Parameter; left = LtNode; right = RtNode; height = 0;
        }
        //Node class members
        AnyType parameter;
        Node<AnyType> left;
        Node<AnyType> right;
        int height;

        //AVL trees cannot have a height imbalance exceeding 1
        private static final int ALLOWED_IMBALANCE = 1;

        private bool IsSubtreeBalanced (Node<AnyType> parameter) {

            boolean balanced = true;

            //base recursion case
            if (Node is null) {
                return 0;
            }
            //compute height for each left subtree node
            int leftNodeHeight = IsSubtreeBalanced(parameter.left);
            //compute height for each right substree node
            int rightNodeHeight = IsSubtreeBalanced(parameter.right);

            if (leftNodeHeight == -1) {
                return false;
            }
            else if (rightNodeHeight == -1) {
                return false;
            }
            if(absolute value of leftNodeHeight - rightNodeHeight > ALLOWED_IMBALANCE) {
                return -1;
            }
            (figure 4.39 as guide)
            //we're taking the max height of the bigger subtree to calculate the height difference, not the smalles
            return Math.max(leftNodeHeight, rightNodeHeight) + 1;
        }
